This document describes how schema that our job representation json should follow:

unionJob -> computeJob | metricsJob

computeJob -> {
    "job_type": COMPUTE,
    "algorithm": ALGORITHM,
    "input": [ NUMBER_LIST_OR_EMPTY ],
    "graph": GRAPH
}

ALGORITHM -> BFS | DFS | DIJKSTRA
GRAPH -> {
    "numVertices": NUMBER,
    "edges": [ GRAPH_ADJ_OR_EMPTY ],
    "vertexLabels": [ LABEL_LIST_OR_EMPTY ],
}

GRAPH_ADJ_OR_EMPTY -> GRAPH_ADJ | EMPTY
NUMBER_LIST_OR_EMPTY -> NUMBER_LIST | EMPTY
LABEL_LIST_OR_EMPTY -> LEBEL_LIST | EMPTY

GRAPH_ADJ -> [ NUMBER_LIST_OR_EMPTY ], GRAPH_ADJ | [ NUMBER_LIST_OR_EMPTY ] 
NUMBER_LIST -> NUMBER, NUMBER_LIST | NUMBER
LABEL_LIST -> STRING, LABEL_LIST | STRING

# LEAVE EMPTY FOR NOW
metricsJob -> {
    "job_type": METRICS
}
